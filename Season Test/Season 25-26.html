// Extract players const players = [...new Set(Object.values(leagueStatsDashboard) .flatMap(spieltag => Object.keys(spieltag)))]; // Spieltage const matchdays = Object.keys(leagueStatsDashboard); // Calculate cumulative points const playerData = {}; players.forEach(player => { let cumulative = 0; playerData[player] = matchdays.map(md => { cumulative += leagueStatsDashboard[md][player] || 0; return cumulative; }); }); // Random colors for players function randomColor() { return `hsl(${Math.floor(Math.random()*360)},70%,50%)`; } // Chart.js datasets const datasets = players.map(player => ({ label: player, data: playerData[player], borderColor: randomColor(), fill: false, tension: 0.2 })); // Render chart new Chart(document.getElementById('fieberChart'), { type: 'line', data: { labels: matchdays, datasets: datasets }, options: { responsive: true, plugins: { legend: { position: 'top', }, title: { display: true, text: 'Fieberkurve (Kumulative Punkte pro Spieltag)' } }, scales: { y: { beginAtZero: true, ticks: { precision:0 } } } } });
